Git Config

[alias]
    co = checkout
    st = status
    ci = commit
    br = branch
    lg = log --graph --decorate --oneline
    lg-detail = log --graph
    review-mr = fetch origin merge-requests/$1/head:mr-$1
    review-br = pull origin rv-$1:$1

[filter "lfs"]
    clean = git-lfs clean -- %f
    smudge = git-lfs smudge -- %f
    process = git-lfs filter-process
    required = true

[user]
    name = A Name
    email = an@email.?

[url "git@github.com:"]
  insteadOf = https://github.com/

[http]
    proxy = http://localhost:9009

[https]
    proxy = http://localhost:9009

Keys for Multiple Github Account

generate keys :

$ ssh-keygen -t rsa

in ~/.ssh/config :

Host user-first
  HostName github.com
  User git
  proxycommand socat - PROXY:localhost:%h:%p,proxyport=9009
  IdentityFile ~/path/to/akey
  IdentitiesOnly yes


Host user-second
  HostName github.com
  User git
  proxycommand socat - PROXY:localhost:%h:%p,proxyport=9009
  IdentityFile ~/path/to/another/key
  IdentitiesOnly yes

then use :

$ git clone git@user-first:organization/project.git
